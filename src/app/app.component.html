ng2-chessboard

Angular2 wrapper for chessboardjs

npm i ng2-chessboard --save

Example


<ng2-chessboard #board
  [(position)]="position"
  [orientation]="orientation"
  [showNotation]="showNotation"
  [draggable]="draggable"
  [animation]="animation">
</ng2-chessboard>

<button (click)="board.move('e2-e4')">Play e2-e4</button>
<button (click)="position = '8/8/4k3/R7/2K5/8/8/8 w -- 1-0'">Change</button>
<button (click)="position = 'start'">Reset</button>
<input type="radio" [(ngModel)]="orientation" [value]="true"> White
<input type="radio" [(ngModel)]="orientation" [value]="false"> Black
<input type="checkbox" [(ngModel)]="showNotation"/> Notation
<input type="checkbox" [(ngModel)]="draggable"/> Draggable
<input type="checkbox" [(ngModel)]="animation"/> Animation
